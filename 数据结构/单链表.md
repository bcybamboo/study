单链表：数据域＋指针域

结构

```
typedef int DataType;

struct Node{
	DataType info;
	struct Node *next;
};
typedef struct Node *LinkList;
```



##### 创建单链表

```
LinkList createNULL(void)
{
	LinkList llist = (LinkList)malloc(sizeof(struct Node));
	if(llist!=NULL)
		llist->next=NULL;
	else
		printf("false!");
	return llist;
}
```

##### 判断单链表是否为空

```
int isNull_List(LinkList llist)
{
	return (llist->next==NULL);
}
```

##### 在单链表中求某元素的存储位置

```
struct Node* locate_link(LinkList llist,DataType x)
{
	struct Node *p;//不能改变head的位置，所以再创建一个节点用来遍历 
	if(llist==NULL)
		return NULL;
	while(p->next!=NULL&&p->info!=x)
	{
		p=p->next;
	}
	return p;//1.找到就返回值 2.没找到就返回NULL 
}
```

##### 单链表的插入

```
int insert(LinkList llist,struct Node *p,DataType x)
{
	struct Node *q = (struct Node *)malloc(sizeof(struct Node));
	if(q==NULL)
		return 0;//插入失败 
	q->info=x;
	q->next=p->next;
	p->next=q;
	return 1;//插入成功 
}
```

##### 求单链表的前驱结点

```
struct Node * locatepre(LinkList llist,struct Node *P)
{
	struct Node *p;
	if(llist==NULL)
		return NULL;
	p=llist;
	while(p!=NULL && p->next!=P)
	{
		p=p->next;
	}
	return p;
}
```

##### 单链表的删除

```
int delete_link(LinkList llist,DataType x)
{
	struct Node *p,*q;
	p=llist;
	if(p==NULL)
		return 0;
	while(p->next!=NULL&&p->next->info!=x)
		p=p->next;
	if(p->next==NULL)
		return 0;
	else
	{
		q=p->next;
		p->next=q->next;
		free(q);
		return 1;
	}
}
```
